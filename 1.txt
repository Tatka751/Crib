git init - ініціалізація в каталозі проекту
git status - переглянути статус файлів.каталогів проекту
git add - додати зміни 
git commit - закомітити зміни
git log - продивитись що відбувалось
git checkout - переміщення між створеними гілками проекту
git branch - створення, перегляд, перейменування, видалення гілки проекту
git push - вивантаження з локального репозиторія у віддалений
